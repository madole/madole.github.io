function Shape(sides, shape) {
    var _sides = sides;
    var _shape = shape;
    
    return {
        makeShape: function(angles) {
            console.log('A ' + _shape + ' has ' + _sides + ' and ' + angles + ' degrees angles');
        }
    }
}

var shape = new Shape(4, 'rectangle');

shape.makeShape(90); // A rectangle has 4 and 90 degrees angles

function doSomething() {
    var x = 10;
    var y = 100;
    
    return {
        multiply: function (a, b) {
            return a * b;
        },
        printToConsole: function (text) {
            console.log(text);
        },
        printAndMultiplyAfterWait: function () {
            setTimeout(function () {
                this.printToConsole(this.multiply(x, y));
            }, 2000);
        }
    };

}

var obj = doSomething();

console.log(obj.multiply(10, 5)); //50

obj.printToConsole(2000); //2000

obj.printAndMultiplyAfterWait(); //undefined is not a function
}

function Person() {
    this.typeOfAnimal = 'human';
}

Person.prototype.speak = function () {
    console.log('Hello');
}

Person.prototype.sleep = function () {
    console.log('Snore...zzz');
}


var andrew = new Person();
console.log(andrew.typeOfAnimal); //human
andrew.speak(); //Hello
andrew.sleep(); //Snore...zzz

var andy = Object.create(Person.prototype); 
console.log(andy.typeOfAnimal); //undefined
andy.speak(); //Hello
andy.sleep(); //Snore...zzz
